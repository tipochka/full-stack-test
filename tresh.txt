    <script>
        $( "#send" ).click(function() {
            var lang = $("select#lang").val();
            var url = 'louder.php?lang='+lang;
            var items = [];
            $.getJSON( url, function( data ) {
                $.each( data, function( key, val ) {
                    items.push( "<div class='" + key + "'>" + val + "</div>" );
                });
                $( items.join( "" )
                ).appendTo( "div.table" );
            });

        });
    </script>
